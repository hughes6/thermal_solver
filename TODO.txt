struct ModelLoader {
    ModelInput model;
    std::unordered_map<std::string, FanCurveInput> fan_curve_library;
    std::unordered_map<std::string, ComponentTemplateInput> component_template_cache;   // NEW

    // ... load_fan_curves, load_model unchanged ...

    const ComponentTemplateInput& get_component_template(const std::string& path) {
        auto it = component_template_cache.find(path);
        if (it != component_template_cache.end()) return it->second;
        auto [inserted, ok] = component_template_cache.emplace(path, load_component_template(path));
        return inserted->second;
    }

    void run() {
        // ... Workload/Environment/Rack/Mesh/Grapher setup unchanged ...

        for (const ComponentInput& c_in : model.components) {
            ComponentInput c = c_in;   // local mutable copy for merging

            if (c.template_path.has_value()) {
                const ComponentTemplateInput& tmpl = get_component_template(*c.template_path);
                if (c.name.empty())              c.name = tmpl.name;
                if (!c.size.has_value())         c.size = tmpl.size;
                if (!c.material.has_value())     c.material = tmpl.material;
                if (!c.watts.has_value())        c.watts = tmpl.watts;
                if (c.internal_regions.empty())  c.internal_regions = tmpl.internal_regions;
            }

            if (c.name.empty()) {
                throw std::runtime_error("Component has no name and no template to supply one");
            }
            if (!c.size.has_value() || !c.material.has_value() || !c.watts.has_value()) {
                throw std::runtime_error("Component '" + c.name + "': missing size/material/watts "
                                          "and no template supplied them");
            }

            const std::string comp_units = c.size->units.value_or("u");
            Component component;
            if(comp_units == "u") {
                component = Component::from_rack_units(c.size->width, c.size->depth, c.size->height, c.name);
            } else if(comp_units == "m") {
                component = Component::from_meters(c.size->width, c.size->depth, c.size->height, c.name);
            } else if(comp_units == "in") {
                component = Component::from_inches(c.size->width, c.size->depth, c.size->height, c.name);
            } else if(comp_units == "mm") {
                component = Component::from_mm(c.size->width, c.size->depth, c.size->height, c.name);
            } else {
                throw std::runtime_error("Invalid component.size units: '" + comp_units + "'.");
            }

            // position dispatch: unchanged from your current run(), still uses c.position directly

            component.set_cp(c.material->cp);
            component.set_rho_solid(c.material->density);
            component.set_k_solid(c.material->k);
            component.set_watts(*c.watts);

            // internal-region loop: unchanged -- i.size.units / i.local_position.units
            // are already "m" for template-sourced regions, "u" (default) for inline ones,
            // so the existing per-instance dispatch handles both transparently.

            component.order_internal_regions();
            mesh.stamp_component(component);
            grapher.add_component(component);
        }

        // ... fans/vents/flow-solver/thermal-solver unchanged ...
    }
};
